Mount route in your API router.

Find the file where module routes are mounted, for example:

backend/src/routes/index.js
or
backend/src/app.js
or
backend/src/server.js

Add:

const flowDiagnosticsRoutes = require('../modules/flow-diagnostics/flowDiagnostics.routes');

Then mount:

router.use('/api', flowDiagnosticsRoutes);

If the router is already mounted under /api, use:

router.use(flowDiagnosticsRoutes);

Example if you have src/routes/index.js:

const express = require('express');
const flowDiagnosticsRoutes = require('../modules/flow-diagnostics/flowDiagnostics.routes');

const router = express.Router();

router.use(flowDiagnosticsRoutes);

module.exports = router;

Test:

curl "http://localhost:4000/api/tokens/COMP/flow-diagnostics?range=1m"
